git cheetsheet

git diff <file>

git log

git push origin main

git status

git branch // check branch

git branch <branch name> // create new branch

git config user.name "name"

git config user.email "email"

git checkout -b <branch name> // create and checkout the branch

git commit -a // stage file automatically

git log -p // produce patch text

git add -p // allow user to interactively review code before commit

git mv

git rm

git reset // unstaged commit

git revert // make changes - roleback previous commit

git merge --abort

GITHUB
------

git remote -v // check remote branch

git remote show origin // show info of the remote repo

git fetch // fetch remote updates but not merge

git pull // fetch remote update and merge

git merge // merge the remote to local manually

git log origin/main

git log -p -1 // check the last commit changes
