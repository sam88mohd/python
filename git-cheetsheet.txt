git cheetsheet

git diff <file>

git log

git remote add origin git@github // add to github

git push origin main

git status

git branch // check branch

git branch <branch name> // create new branch

git config user.name "name"

git config user.email "email"

git checkout -b <branch name> // create and checkout the branch

git commit -a // stage file automatically

git log -p // produce patch text

git add -p // allow user to interactively review code before commit

git mv

git rm

git reset // unstaged commit

git revert // make changes - roleback previous commit

git merge --abort

GITHUB
------

git remote -v // check remote branch

git remote show origin // show info of the remote repo

git fetch // fetch remote updates but not merge

git pull // fetch remote update and merge

git merge // merge the remote to local manually

git log origin/main

git log -p -1 // check the last commit changes

git log --graph --oneline --all // show different commit and position in the tree // usefull if commit failed

MERGE BRANCH TO MASTER BRANCH // move current branch on top of branch

1. git checkout <branch>

2. git rebase master

3. git checkout master

4. git merge <branch>

git push --delete origin <branch> // delete remote branch

git branch -d <branch>

git push -f // if using to submit pull request // dont use in public repo

git commit -a --ammend // make changes t previous commit // return new commit with new commit ID // use with push -f
